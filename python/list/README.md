# list

```python
mylist = ["apple", "banana", "cherry"]  
                          # ['apple', 'banana', 'cherry']

mylist = [1, 3, 5]        # [1, 3, 5]

mylist = list(1, 3, 5)    # ERROR! 

mylist = list((1, 3, 5))  # !!! [1, 3, 5]

mylist = []               # []

mylist = list()           # []

mylist = list("abra")     # !!! ['a', 'b', 'r', 'a']

mylist = list(("abra"))   # !!! ['a', 'b', 'r', 'a']

mylist = list(("abra",))  # !!! ['abra']
```
